// --- Helper Functions ---
const calculateScore = (answers) => {
  let total = 0;
  answers.forEach((ans) => {
    const question = QUESTIONS.find(q => q.id === ans.questionId);
    if (question.reverse) {
      // Reverse scoring: 1=5, 2=4, 3=3, 4=2, 5=1
      total += (6 - ans.value);
    } else {
      total += ans.value;
    }
  });
  return total;
};

// Updated Colors for Pink Theme
// High = Pink (Love), Average = Purple, Low = Gray (No spark)
const getAssessmentResult = (score) => {
  if (score >= 22) return { level: 'High Satisfaction', color: 'text-pink-600', bg: 'bg-pink-100', border: 'border-pink-200' };
  if (score >= 15) return { level: 'Average Satisfaction', color: 'text-purple-600', bg: 'bg-purple-100', border: 'border-purple-200' };
  return { level: 'Low Satisfaction', color: 'text-slate-600', bg: 'bg-slate-100', border: 'border-slate-200' };
};

const getCoachTip = (score) => {
  if (score >= 22) return "Your relationship is thriving! Maintain this momentum by introducing novel activities to keep the excitement alive. Focus on 'capitalization'â€”celebrating each other's small wins enthusiastically.";
  if (score >= 15) return "You have a solid foundation. To move from average to excellent, try a weekly 'check-in' where you discuss one unmet need without judgment. Focus on appreciation over criticism.";
  return "Relationships go through seasons. Right now, focus on the basics: reliable communication and meeting simple tangible needs. Consider professional counseling to help navigate these blocks safely.";
};

const Assessment = ({ onComplete, onCancel }) => {
  const [currentStep, setCurrentStep] = useState(0);
  const [answers, setAnswers] = useState([]);

  const handleAnswer = (value) => {
    const newAnswers = [...answers, { questionId: QUESTIONS[currentStep].id, value }];
    if (currentStep < QUESTIONS.length - 1) {
      setAnswers(newAnswers);
      setCurrentStep(currentStep + 1);
    } else {
      onComplete(newAnswers);
    }
  };

  const question = QUESTIONS[currentStep];
  const progress = ((currentStep + 1) / QUESTIONS.length) * 100;

  return (
    <div className="fixed inset-0 bg-white z-50 flex flex-col" style={patternStyle}>
      {/* Progress Bar */}
      <div className="h-2 bg-pink-100">
        <div 
          className="h-full bg-pink-500 transition-all duration-500 ease-out shadow-[0_0_10px_rgba(236,72,153,0.5)]"
          style={{ width: `${progress}%` }}
        />
      </div>

      <div className="flex-1 flex flex-col items-center justify-center p-6 max-w-2xl mx-auto w-full">
        <div className="bg-white/90 p-8 rounded-3xl shadow-xl border border-pink-100 backdrop-blur-md w-full">
          <div className="mb-10 text-center">
            <span className="text-pink-500 font-bold tracking-wider text-xs uppercase mb-3 block bg-pink-50 inline-block px-3 py-1 rounded-full">
              Question {currentStep + 1} of {QUESTIONS.length}
            </span>
            <h2 className="text-2xl md:text-3xl font-bold text-slate-800 leading-tight">
              {question.text}
            </h2>
          </div>

          <div className="w-full space-y-6">
            <div className="flex justify-between text-sm font-semibold text-slate-400 px-1">
              <span>{question.lowLabel}</span>
              <span>{question.highLabel}</span>
            </div>
            
            <div className="grid grid-cols-5 gap-3">
              {[1, 2, 3, 4, 5].map((val) => (
                <button
                  key={val}
                  onClick={() => handleAnswer(val)}
                  className="aspect-square rounded-2xl border-2 border-pink-100 hover:border-pink-500 hover:bg-pink-500 hover:text-white text-xl font-bold text-pink-400 transition-all focus:outline-none focus:ring-4 focus:ring-pink-200 transform hover:scale-105 shadow-sm"
                >
                  {val}
                </button>
              ))}
            </div>
            <div className="text-center text-xs text-slate-400 font-medium">
              Select 1 (Lowest) to 5 (Highest)
            </div>
          </div>
        </div>
      </div>

      <div className="p-6 flex justify-center bg-white/50 backdrop-blur-sm">
        <button
          onClick={onCancel}
          className="text-slate-400 hover:text-slate-600 font-medium text-sm transition-colors flex items-center hover:bg-slate-100 px-4 py-2 rounded-lg"
        >
          <LogOut className="w-4 h-4 mr-2" />

      </div>
    </div>
  );
};

export default function App() {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [coupleId, setCoupleId] = useState(() => localStorage.getItem('ras_coupleId') || '');
  const [partnerName, setPartnerName] = useState(() => localStorage.getItem('ras_partnerName') || '');
  const [view, setView] = useState('dashboard'); // 'dashboard' or 'assessment'
  const [assessments, setAssessments] = useState([]);

  // --- Auth & Initial Load ---
  useEffect(() => {
    const initAuth = async () => {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };

    initAuth();
    const unsubscribe = onAuthStateChanged(auth, (currentUser) => {
      setUser(currentUser);
      setLoading(false);
    });
    return () => unsubscribe();
  }, []);

  // --- Data Fetching (Rule 2 Compliant) ---
  useEffect(() => {
    if (!user || !coupleId) return;

    // Fetch from public collection, we will filter in memory
    const q = query(collection(db, 'artifacts', appId, 'public', 'data', 'ras-assessments'));
    
    const unsubscribe = onSnapshot(q, 
      (snapshot) => {
        const loadedData = snapshot.docs.map(doc => ({
          id: doc.id,
          ...doc.data()
        }));
        setAssessments(loadedData);
      },
      (error) => {
        console.error("Error fetching data:", error);
      }
    );

    return () => unsubscribe();
  }, [user, coupleId]);

  const handleJoin = (cId, pName) => {
    localStorage.setItem('ras_coupleId', cId);
    localStorage.setItem('ras_partnerName', pName);
    setCoupleId(cId);
    setPartnerName(pName);
  };

  const handleLogout = async () => {
    localStorage.removeItem('ras_coupleId');
    localStorage.removeItem('ras_partnerName');
    setCoupleId('');
    setPartnerName('');
  };

  const handleAssessmentComplete = async (answers) => {
    const score = calculateScore(answers);
    const today = new Date().toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
    
    try {
      await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'ras-assessments'), {
        coupleId,
        partnerName,
        score,
        answers,
        date: today,
        timestamp: serverTimestamp()
      });
      setView('dashboard');
    } catch (err) {
      console.error("Error saving assessment", err);
    }
  };

  if (loading) return <LoadingScreen />;

  if (!coupleId) return <LandingPage onJoin={handleJoin} />;

  if (view === 'assessment') {
    return (
      <Assessment 
        onComplete={handleAssessmentComplete} 
        onCancel={() => setView('dashboard')} 
      />
    );
  }

  return (
    <Dashboard 
      coupleId={coupleId}
      partnerName={partnerName}
      data={assessments}
      onStartAssessment={() => setView('assessment')}
      onLogout={handleLogout}
    />
  );
}
